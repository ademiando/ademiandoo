        const isTouchDevice = 'ontouchstart' in window;
        if (isTouchDevice) {

        }

        function createRipple(x, y) {
            const rippleContainer = document.createElement('div');
            rippleContainer.style.position = 'fixed';
            rippleContainer.style.left = `${x}px`;
            rippleContainer.style.top = `${y}px`;
            rippleContainer.style.zIndex = '1000';
            rippleContainer.style.pointerEvents = 'none';
            
            const ripple = document.createElement('div');
            ripple.className = 'ripple';
            rippleContainer.appendChild(ripple);
            document.body.appendChild(rippleContainer);
            
            setTimeout(() => {
                rippleContainer.remove();
            }, 1200);
        }

        if (!isTouchDevice) {
            cursor.classList.add('visible');
            window.addEventListener('mousemove', (e) => {
                cursor.style.left = `${e.clientX}px`;
                cursor.style.top = `${e.clientY}px`;
            });
        } else {
            document.addEventListener('touchstart', (e) => {
                const touch = e.touches[0];
                createRipple(touch.clientX, touch.clientY);
            }, {passive: true});
        }
        
        const interactiveElements = document.querySelectorAll('a, button, .menu-btn-wrapper');
        interactiveElements.forEach(el => {
            el.addEventListener('mouseenter', () => {
                if (!isTouchDevice) {
                    cursor.classList.add('hovered');
                    const newRipple = document.createElement('div');
                    newRipple.className = 'ripple';
                    cursor.appendChild(newRipple);
                    setTimeout(() => newRipple.remove(), 1200);
                }
            });

        // --- WebGL Background Logic ---
        let scene, camera, renderer;
        const clock = new THREE.Clock();
        let animationLoop = () => {};

        function initBaseScene() {
            scene = new THREE.Scene();
            camera = new THREE.PerspectiveCamera(75, window.innerWidth / window.innerHeight, 0.1, 1000);
            const canvas = document.getElementById('bg-canvas');
            renderer = new THREE.WebGLRenderer({
                canvas: canvas,
                antialias: true,
                alpha: true
            });
            renderer.setSize(window.innerWidth, window.innerHeight);
            renderer.setPixelRatio(Math.min(window.devicePixelRatio, 2));
            window.addEventListener('resize', onWindowResize);
        }

        function onWindowResize() {
            camera.aspect = window.innerWidth / window.innerHeight;
            camera.updateProjectionMatrix();
            renderer.setSize(window.innerWidth, window.innerHeight);
        }

        function createMonolithScene() {
            scene.fog = new THREE.Fog(0x0A0A12, 5, 15);
            camera.position.z = 7;
            const geometry = new THREE.BoxGeometry(0.7, 5, 0.7);
            const material = new THREE.MeshStandardMaterial({ color: 0x111827, metalness: 0.95, roughness: 0.3 });
            const monolith = new THREE.Mesh(geometry, material);
            scene.add(monolith);
            const spotlight = new THREE.SpotLight(0xffffff, 5, 30, Math.PI * 0.1, 0.4, 1.5);
            spotlight.position.set(0, 5, 5);
            scene.add(spotlight);
            scene.add(new THREE.AmbientLight(0xffffff, 0.05));
            animationLoop = () => {
                const t = clock.getElapsedTime();
                monolith.rotation.y = t * 0.08;
                monolith.rotation.x = t * 0.02;
                spotlight.position.x = Math.sin(t * 0.3) * 5;
                spotlight.position.z = Math.cos(t * 0.3) * 5;
                spotlight.target = monolith;
            };
        }

        function createTectonicScene() {
            camera.position.set(0, 4, 6);
            camera.lookAt(0, 0, 0);
            scene.fog = new THREE.FogExp2(0x0A0A12, 0.08);
            const geometry = new THREE.PlaneGeometry(30, 30, 150, 150);
            const material = new THREE.MeshStandardMaterial({ color: 0x030712, metalness: 0.8, roughness: 0.6 });
            const plane = new THREE.Mesh(geometry, material);
            plane.rotation.x = -Math.PI / 2;
            scene.add(plane);
            const originalPositions = new Float32Array(geometry.attributes.position.array);
            const light1 = new THREE.PointLight(0x5b21b6, 3, 15);
            light1.position.set(-8, 2, 0);
            scene.add(light1);
            const light2 = new THREE.PointLight(0x155e75, 3, 15);
            light2.position.set(8, 2, 0);
            scene.add(light2);
            animationLoop = () => {
                const t = clock.getElapsedTime();
                const positions = geometry.attributes.position.array;
                for (let i = 0; i < positions.length / 3; i++) {
                    const x = originalPositions[i * 3], y = originalPositions[i * 3 + 1];
                    const wave1 = Math.sin(x * 0.4 + t * 0.5) * 0.4, wave2 = Math.sin(y * 0.3 + t * 0.3) * 0.4;
                    positions[i * 3 + 2] = wave1 + wave2;
                }
                geometry.attributes.position.needsUpdate = true;
                light1.position.x = Math.sin(t * 0.5) * 8;
                light2.position.x = Math.cos(t * 0.3) * -8;
            };
        }
        
        function createRingsScene() {
            camera.position.set(0, 3, 7);
            camera.lookAt(0, 0, 0);
            const ringsGroup = new THREE.Group();
            scene.add(ringsGroup);
            const createRing = (radius) => {
                const ring = new THREE.Mesh(new THREE.TorusGeometry(radius, 0.025, 24, 128), new THREE.MeshStandardMaterial({ color: 0x6b7280, metalness: 1.0, roughness: 0.2 }));
                ring.rotation.x = Math.PI / 2;
                return ring;
            };
            for(let i = 0; i < 5; i++) ringsGroup.add(createRing(2.5 + i * 0.5));
            const keyLight = new THREE.SpotLight(0xffffff, 8, 30, Math.PI * 0.1, 0.2, 2);
            keyLight.position.set(0, 8, 4);
            scene.add(keyLight);
            scene.add(new THREE.AmbientLight(0xffffff, 0.1));
            animationLoop = () => {
                const t = clock.getElapsedTime();
                ringsGroup.children.forEach((ring, i) => {
                    ring.rotation.z = t * (0.1 - i * 0.02) * (i % 2 === 0 ? 1 : -1);
                });
                ringsGroup.rotation.y = t * 0.04;
                ringsGroup.rotation.z = t * 0.02;
            };
        }

        function animate() {
            requestAnimationFrame(animate);
            animationLoop();
            renderer.render(scene, camera);
        }

        initBaseScene();
        const scenes = [createMonolithScene, createTectonicScene, createRingsScene];
        scenes[Math.floor(Math.random() * scenes.length)]();
        animate();
    </script>
